 \newcommand{\Ais}{
 The components of the vector potential, ${\bf A}=A_\t \nabla + A_\z \nabla \z$, are
 \be A_\t(\s,\t,\z) &=& \sum_{i,l} \Ate{i,l} \; \bT{l,i}(s) \ci + \sum_{i,l} \Ato{i,l} \; \bT{l,i}(s) \si, \label{eq:At} \\
     A_\z(\s,\t,\z) &=& \sum_{i,l} \Aze{i,l} \; \bT{l,i}(s) \ci + \sum_{i,l} \Azo{i,l} \; \bT{l,i}(s) \si, \label{eq:Az}
 \ee
 where $\bT{l,i}(s) \equiv \bar s^{m_i/2} \, T_l(s)$, $T_l(s)$ is the Chebyshev polynomial, and $\a_j \equiv m_j\t-n_j\z$.
 The regularity factor, $\bar s^{m_i/2}$, where $\bar s \equiv (1+s)/2$, is only included if there is a coordinate singularity in the domain
 (i.e. only in the innermost volume, and only in cylindrical and toroidal geometry.)
 }

 \newcommand{\Bis}{
 The magnetic field, $\sqrt g \, {\bf B} = \sqrt g B^\s {\bf e}_\s + \sqrt g B^\t {\bf e}_\t + \sqrt g B^\z {\bf e}_\z$, is
 \be \begin{array}{ccccrcrcrcrcccccccccccccccccccccccccccccccccccccccccccccccccccc}
 \!\!\!\!\!\!\!\!\!\! \sqrt g \, {\bf B} & = & {\bf e}_\s & \sum_{i,l} [ 
 ( & - m_i \Aze{i,l} & - & n_i \Ate{i,l} & ) \bT{l,i}  \si + ( & + m_i \Azo{i,l} & + & n_i \Ato{i,l} & ) \bT{l,i} \ci
 ] \\
                    & + & {\bf e}_\t & \sum_{i,l} [ 
 ( &                 & - &     \Aze{i,l} & ) \bT{l,i}' \ci + ( &                 & - &     \Azo{i,l} & ) \bT{l,i}' \si
 ] \\
                    & + & {\bf e}_\z & \sum_{i,l} [ 
 ( &       \Ate{i,l} &   &               & ) \bT{l,i}' \ci + ( &       \Ato{i,l} &   &               & ) \bT{l,i}' \si
 ] 
 \end{array}
 \ee
 }
